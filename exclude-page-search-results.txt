function redirect_to_post_on_1_search_result() {
    if (is_search()) {
        global $wp_query;
        if ($wp_query->post_count == 1) {
            wp_redirect( get_permalink( $wp_query->posts['0']->ID ) );
        }
    }
}
add_action('template_redirect', 'redirect_to_post_on_1_search_result');
